```sql
  family_ids      TEXT NOT NULL,                 -- JSON array de family_ids del drop
  created_at      INTEGER NOT NULL DEFAULT (unixepoch())
);

CREATE INDEX IF NOT EXISTS idx_thematic_active ON thematic_drops(publish_at, unpublish_at);
```

## Algoritmo: pseudocódigo completo

### `select_mystery_for_customer(customer_id) → family_id`

```python
def select_mystery_for_customer(customer_id: int) -> str:
    """
    Selecciona una jersey del pool Mystery para un cliente específico.
    Aplica anti-repeat, fairness boost, exclusiones, y modo de drops temáticos.

    Returns: family_id de la jersey elegida.
    Raises: PoolEmptyError si no hay candidatos válidos.
    """
    rules = load_mystery_rules()  # de admin_web_config

    # === 1. POOL BASE ===
    pool = query_active_pool()
    # SQL: SELECT family_id, pool_weight FROM v_mystery_status
    #      WHERE computed_status = 'live'
    if not pool:
        raise PoolEmptyError()

    # === 2. APLICAR EXCLUSIONES AUTOMÁTICAS ===
    if rules['exclude_low_coverage']:
        pool = [j for j in pool if j.coverage >= 5]
    if rules['exclude_dirty']:
        pool = [j for j in pool if not j.dirty_flag]
    if rules['exclude_supplier_gap']:
        pool = [j for j in pool if not j.supplier_gap]

    # === 3. APLICAR DROP TEMÁTICO ACTIVO (si existe) ===
    active_drop = query_active_thematic_drop()
    if active_drop:
        if active_drop.mode == 'substitute':
            # Pool restringido SOLO al subset del drop
            drop_family_ids = json.loads(active_drop.family_ids)
            pool = [j for j in pool if j.family_id in drop_family_ids]
        elif active_drop.mode == 'restrict':
            # Pool normal + bias fuerte hacia las del drop
            for j in pool:
                if j.family_id in json.loads(active_drop.family_ids):
                    j.effective_weight = j.pool_weight * active_drop.bias_multiplier
                else:
                    j.effective_weight = j.pool_weight
        else:
            for j in pool:
                j.effective_weight = j.pool_weight
    else:
        for j in pool:
            j.effective_weight = j.pool_weight

    # === 4. ANTI-REPEAT (si activo) ===
    if rules['anti_repeat_enabled']:
        cooldown_days = rules['anti_repeat_cooldown_days']
        cooldown_ts = now() - (cooldown_days * 86400)
        recent_deliveries = query_recent_deliveries(customer_id, since=cooldown_ts)
        # SQL: SELECT family_id FROM mystery_deliveries
        #      WHERE customer_id = ? AND delivered_at > ?
        recent_family_ids = {d.family_id for d in recent_deliveries}
        pool = [j for j in pool if j.family_id not in recent_family_ids]

        if not pool:
            # Anti-repeat dejó pool vacío: log warning + relax temporalmente
            log_warning(f"Anti-repeat exhausted for customer {customer_id}, relaxing")
            pool = query_active_pool()  # reset, ignora anti-repeat este round

    # === 5. FAIRNESS BOOST (si activo) ===
    if rules['fairness_never_delivered']:
        for j in pool:
            never_delivered = query_delivery_count(j.family_id) == 0
            if never_delivered:
                j.effective_weight *= 2.0  # ⭐ BOOST nunca-entregada

    if rules['fairness_low_deliveries']:
        avg_deliveries = query_avg_deliveries_pool()
        for j in pool:
            j_deliveries = query_delivery_count(j.family_id)
            if j_deliveries < (avg_deliveries * 0.3):
                j.effective_weight *= 1.5  # ⭐ BOOST low-deliveries

    # === 6. APLICAR PREFERENCIAS DEL CLIENTE (si modo lo requiere) ===
    if rules['algorithm'] in ('preferences', 'hybrid'):
        prefs = query_customer_preferences(customer_id)
        # prefs viene del wizard de checkout: equipos a evitar, jugadores a evitar
        if prefs.avoided_teams:
            pool = [j for j in pool if j.team not in prefs.avoided_teams]
        if prefs.avoided_player_jerseys:  # ej. específicamente jerseys de Messi
            pool = [j for j in pool if (j.team, j.season) not in prefs.avoided_player_jerseys]

        if not pool:
            log_warning(f"Customer prefs left pool empty, ignoring prefs")
            pool = query_active_pool()  # fallback

    # === 7. SELECCIÓN FINAL SEGÚN ALGORITMO ===
    if rules['algorithm'] == 'uniform':
        chosen = random.choice(pool)
    elif rules['algorithm'] in ('weighted', 'preferences', 'hybrid'):
        # Random ponderado por effective_weight
        weights = [j.effective_weight for j in pool]
        chosen = random.choices(pool, weights=weights, k=1)[0]

    # === 8. PERSIST DELIVERY ===
    insert_delivery(
        family_id=chosen.family_id,
        customer_id=customer_id,
        order_id=current_order_id,  # del context
        algorithm_used=rules['algorithm'],
        weight_at_pick=chosen.effective_weight,
        pool_size_at_pick=len(pool)
    )

    # === 9. RETURN ===
    return chosen.family_id
```

### Sub-funciones helper

```python
def query_active_pool():
    """SELECT family_id, pool_weight, coverage, dirty_flag, supplier_gap, team, season
       FROM v_mystery_status JOIN audit_decisions USING(family_id)
       WHERE computed_status = 'live'"""

def query_active_thematic_drop():
    """SELECT * FROM thematic_drops
       WHERE publish_at <= ? AND (unpublish_at IS NULL OR unpublish_at > ?)
       LIMIT 1"""

def query_recent_deliveries(customer_id, since):
    """SELECT family_id FROM mystery_deliveries
       WHERE customer_id = ? AND delivered_at > ?"""

def query_delivery_count(family_id):
    """SELECT COUNT(*) FROM mystery_deliveries WHERE family_id = ?"""

def query_avg_deliveries_pool():
    """SELECT AVG(c) FROM (
         SELECT COUNT(*) c FROM mystery_deliveries
         WHERE family_id IN (SELECT family_id FROM v_mystery_status WHERE computed_status='live')
         GROUP BY family_id
       )"""

def query_customer_preferences(customer_id):

       o lee del wizard intermediario que ya guardamos en current order.
       Returns: { avoided_teams: [...], avoided_player_jerseys: [...] }"""
```

### Edge cases

```python
# CASE 1: Pool empty after all filters
# → Insert inbox event 'mystery_pool_empty'
# → Fallback: customer service manual delivery
# → Return None, frontend muestra "Estamos curando tu sorpresa, te contactamos"

# CASE 2: Customer has anti-repeat exhausted (todas las jerseys del pool ya entregadas)
# → log_warning + relax cooldown temporalmente
# → Considera escalar a Diego después de N customers afectados

# CASE 3: Drop temático arranca/termina mid-selection
# → Snapshot del pool al inicio de la selección, no re-query
# → Race condition no aplica porque la selección es atómica

# CASE 4: Customer preferences contradicen pool entero
# → Fallback ignora prefs, log warning, customer recibe sorpresa con team que pidió evitar
# → Inbox event para Diego: "Customer X recibió jersey que pidió evitar"
```

## Riesgos identificados

1. **Wizard de checkout existente** — necesita auditar cómo captura prefs y si están persistidas. Si no, agregar el flujo es un sub-projecto.

2. **Anti-repeat exhaustion** — pool chico (<20 jerseys) + cooldown alto (60d) puede dejar customers sin opciones. Mitigación: relax automático + alert.

3. **Fairness boost loop** — si jersey nunca-entregada tiene weight×2, después la primera entrega ya no aplica. Verificar que el boost se calcula al inicio de cada selección, no se persiste.

4. **Thematic drops switching** — race condition si drop arranca/termina durante selección concurrente. Snapshot al inicio mitiga.

5. **Cross-table queries pesadas** — métricas avanzadas pueden ser lentas con 1000+ deliveries. Cachear a `kpi_snapshots`.
